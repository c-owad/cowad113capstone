* import React, { useState, useEffect } from 'react';
import { 
  StyleSheet, 
  Text, 
  View, 
  TextInput, 
  TouchableOpacity, 
  Alert, 
  SafeAreaView 
} from 'react-native';
import AsyncStorage from '@react-native-async-storage/async-storage';

// --- Expanded Sentence Bank ---
const TARGET_SENTENCES = [
  "The quick brown fox jumps over the lazy dog.",
  "React Native makes building mobile apps intuitive.",
  "Coding is the art of solving puzzles with logic.",
  "AsyncStorage persists data across app restarts.",
  "How fast can your fingers dance across the screen?",
  "Pack my box with five dozen liquor jugs.",
  "Jinxed wizards pluck ivy from the big quilt.",
  "A journey of a thousand miles begins with a single step.",
  "There is nothing permanent except change.",

  "The only impossible journey is the one you never begin.",

  "Success is not final, failure is not fatal: it is the courage to continue that counts.",
  "Software and cathedrals are much the same; first we build them, then we pray."
];

export default function RapidTypeGame() {
  // --- Game State ---
  const [currentScreen, setCurrentScreen] = useState('Menu'); 
  const [highScore, setHighScore] = useState(0);
  const [allowErrors, setAllowErrors] = useState(false); // Toggle for Forgiving Mode
  
  // --- Active Game State ---
  const [targetSentence, setTargetSentence] = useState('');
  const [inputText, setInputText] = useState('');
  const [startTime, setStartTime] = useState(null);

  // Load high score when returning to the Menu
  useEffect(() => {
    if (currentScreen === 'Menu') {
      const fetchHighScore = async () => {
        try {
          const storedScore = await AsyncStorage.getItem('@rapid_type_highscore');
          if (storedScore !== null) {
            setHighScore(parseInt(storedScore, 10));
          }
        } catch (e) {
          console.error("Failed to load high score.");
        }
      };
      fetchHighScore();
    }
  }, [currentScreen]);

  // --- Actions ---
  const startGame = () => {
    const randomIndex = Math.floor(Math.random() * TARGET_SENTENCES.length);
    setTargetSentence(TARGET_SENTENCES[randomIndex]);
    setInputText('');
    setStartTime(null);
    setCurrentScreen('Game');
  };

  const handleTextChange = async (text) => {
    // Start timer on first keystroke
    if (!startTime && text.length > 0) {
      setStartTime(Date.now());
    }

    setInputText(text);

    // Win condition check based on current mode
    if (allowErrors) {
      // In Forgiving Mode, finish when they reach the sentence length
      if (text.length >= targetSentence.length) {
        finishGame(text.substring(0, targetSentence.length));
      }
    } else {
      // In Perfect Mode, finish only when it matches perfectly
      if (text === targetSentence) {
        finishGame(text);
      }
    }
  };

  const finishGame = async (finalText) => {
    const endTime = Date.now();
    const timeInMinutes = (endTime - startTime) / 60000;
    
    const wordCount = targetSentence.length / 5;
    const wpm = Math.round(wordCount / timeInMinutes);

    // Calculate errors
    let errorCount = 0;
    for (let i = 0; i < finalText.length; i++) {
      if (finalText[i] !== targetSentence[i]) {
        errorCount++;
      }
    }

    handleWin(wpm, errorCount);
  };

  const handleWin = async (wpm, errorCount) => {
    try {
      const storedScore = await AsyncStorage.getItem('@rapid_type_highscore');
      const currentHigh = storedScore !== null ? parseInt(storedScore, 10) : 0;

      let message = `Speed: ${wpm} WPM`;
      
      if (allowErrors) {
        message += `\nErrors: ${errorCount}`;
      }

      if (wpm > currentHigh) {
        await AsyncStorage.setItem('@rapid_type_highscore', wpm.toString());
        message += "\n\n🎉 NEW HIGH SCORE! 🎉";
      }

      Alert.alert(
        "Complete!",
        message,
        [{ text: "Awesome", onPress: () => setCurrentScreen('Menu') }],
        { cancelable: false }
      );
    } catch (e) {
      console.error("Failed to save score.");
      setCurrentScreen('Menu');
    }
  };

  // --- Render Helpers ---
  const renderTargetText = () => {
    return (
      <Text style={styles.targetSentenceBox}>
        {targetSentence.split('').map((char, index) => {
          let color = '#333';
          let backgroundColor = 'transparent';
          
          if (index < inputText.length) {
            const isCorrect = inputText[index] === char;
            color = isCorrect ? '#4CAF50' : '#FFF'; 
            backgroundColor = isCorrect ? 'transparent' : '#F44336'; // Highlight typos with red background
          }
          return (
            <Text key={index} style={{ color, backgroundColor }}>
              {char}
            </Text>
          );
        })}
      </Text>
    );
  };

  // --- Screen Renders ---
  if (currentScreen === 'Menu') {
    return (
      <SafeAreaView style={styles.container}>
        <View style={styles.menuContent}>
          <Text style={styles.title}>RAPID TYPE</Text>
          <Text style={styles.scoreText}>All-Time High Score: {highScore} WPM</Text>
          
          <TouchableOpacity style={styles.startButton} onPress={startGame}>
            <Text style={styles.startButtonText}>START GAME</Text>
          </TouchableOpacity>

          {/* Mode Toggle Button */}
          <TouchableOpacity 
            style={styles.modeButton} 
            onPress={() => setAllowErrors(!allowErrors)}
          >
            <Text style={styles.modeButtonText}>
              Mode: {allowErrors ? 'Forgiving (Allows Typos)' : 'Perfect (Strict)'}
            </Text>
          </TouchableOpacity>
        </View>
      </SafeAreaView>
    );
  }

  return (
    <SafeAreaView style={styles.container}>
      <View style={styles.gameContent}>
        {renderTargetText()}
        
        <TextInput
          style={styles.input}
          placeholder="Start typing here..."
          value={inputText}
          onChangeText={handleTextChange}
          autoFocus={true}
          autoCorrect={false}
          autoCapitalize="none"
          maxLength={allowErrors ? targetSentence.length : undefined} // Prevent over-typing in Forgiving mode
        />

        <TouchableOpacity 
          style={styles.cancelButton} 
          onPress={() => setCurrentScreen('Menu')}
        >
          <Text style={styles.cancelButtonText}>Give Up</Text>
        </TouchableOpacity>
      </View>
    </SafeAreaView>
  );
}
